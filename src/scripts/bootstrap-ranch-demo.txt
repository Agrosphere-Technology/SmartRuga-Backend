function makeApi(token: string) {
    return axios.create({
        baseURL: BASE_URL,
        headers: {
            Authorization: `Bearer ${token}`,
            "Content-Type": "application/json",
        },
    });
}

async function getRanchMembers(token: string, ranchSlug: string) {
    const api = makeApi(token);
    const res = await api.get(`/ranches/${ranchSlug}/members`);
    return res.data?.data?.members ?? [];
}

async function createAnimals(ownerToken: string, ranch: RanchConfig) {
    const api = makeApi(ownerToken);
    const created: any[] = [];

    for (const animal of ranch.animals) {
        try {
            const res = await api.post(`/ranches/${ranch.slug}/animals`, animal);
            const createdAnimal = res.data?.data?.animal;
            created.push(createdAnimal);
            console.log(`✅ [${ranch.slug}] Animal created: ${animal.tagNumber}`);
        } catch (err: any) {
            console.log(
                `❌ [${ranch.slug}] Failed animal ${animal.tagNumber}:`,
                err.response?.data ?? err.message
            );
        }
    }

    return created;
}

async function createHealthRecords(vetToken: string, ranch: RanchConfig, animals: any[]) {
    const api = makeApi(vetToken);

    const records = [
        {
            animalPublicId: animals[0]?.publicId ?? animals[0]?.public_id,
            payload: {
                status: "healthy",
                notes: "Routine veterinary inspection completed.",
            },
        },
        {
            animalPublicId: animals[1]?.publicId ?? animals[1]?.public_id,
            payload: {
                status: "sick",
                notes: "Observed reduced appetite and mild weakness.",
            },
        },
        {
            animalPublicId: animals[2]?.publicId ?? animals[2]?.public_id,
            payload: {
                status: "recovering",
                notes: "Responding positively to treatment and observation.",
            },
        },
    ].filter((r) => r.animalPublicId);

    for (const record of records) {
        try {
            await api.post(
                `/ranches/${ranch.slug}/animals/${record.animalPublicId}/health-events`,
                record.payload
            );
            console.log(
                `✅ [${ranch.slug}] Health record created for ${record.animalPublicId}`
            );
        } catch (err: any) {
            console.log(
                `❌ [${ranch.slug}] Failed health record for ${record.animalPublicId}:`,
                err.response?.data ?? err.message
            );
        }
    }
}

async function createTasks(
    ranch: RanchConfig,
    ownerToken: string,
    managerToken: string,
    members: any[]
) {
    const ownerApi = makeApi(ownerToken);
    const managerApi = makeApi(managerToken);

    const workerMember = members.find(
        (m: any) => m.ranchRole === "worker" || m.role === "worker"
    );
    const vetMember = members.find(
        (m: any) => m.ranchRole === "vet" || m.role === "vet"
    );
    const storekeeperMember = members.find(
        (m: any) => m.ranchRole === "storekeeper" || m.role === "storekeeper"
    );

    const resolveUserPublicId = (member: any) =>
        member?.user?.publicId ??
        member?.user?.public_id ??
        member?.user?.id ??
        member?.publicId ??
        member?.id;

    const ownerTasks = [
        workerMember && {
            title: `Daily livestock inspection - ${ranch.slug}`,
            description: "Inspect the main livestock sections and report observations.",
            assignedToUserPublicId: resolveUserPublicId(workerMember),
            dueDate: "2026-04-25",
        },
        storekeeperMember && {
            title: `Check feed availability - ${ranch.slug}`,
            description: "Verify store balance and report any shortages.",
            assignedToUserPublicId: resolveUserPublicId(storekeeperMember),
            dueDate: "2026-04-25",
        },
    ].filter(Boolean);

    const managerTasks = [
        vetMember && {
            title: `Review sick animals - ${ranch.slug}`,
            description: "Check animals with health concerns and update records.",
            assignedToUserPublicId: resolveUserPublicId(vetMember),
            dueDate: "2026-04-24",
        },
        workerMember && {
            title: `Clean holding area - ${ranch.slug}`,
            description: "Clean and sanitize the designated holding area.",
            assignedToUserPublicId: resolveUserPublicId(workerMember),
            dueDate: "2026-04-24",
        },
    ].filter(Boolean);

    for (const task of ownerTasks as any[]) {
        try {
            await ownerApi.post(`/ranches/${ranch.slug}/tasks`, task);
            console.log(`✅ [${ranch.slug}] Owner created task: ${task.title}`);
        } catch (err: any) {
            console.log(
                `❌ [${ranch.slug}] Owner failed task ${task.title}:`,
                err.response?.data ?? err.message
            );
        }
    }

    for (const task of managerTasks as any[]) {
        try {
            await managerApi.post(`/ranches/${ranch.slug}/tasks`, task);
            console.log(`✅ [${ranch.slug}] Manager created task: ${task.title}`);
        } catch (err: any) {
            console.log(
                `❌ [${ranch.slug}] Manager failed task ${task.title}:`,
                err.response?.data ?? err.message
            );
        }
    }
}

async function createConcern(workerToken: string, ranch: RanchConfig) {
    const api = makeApi(workerToken);

    try {
        await api.post(`/ranches/${ranch.slug}/concerns`, {
            title: `Feed level concern - ${ranch.slug}`,
            description:
                "Observed feed quantity dropping faster than expected in one section.",
            category: "inventory",
            priority: "high",
        });
        console.log(`✅ [${ranch.slug}] Worker raised concern`);
    } catch (err: any) {
        console.log(
            `❌ [${ranch.slug}] Failed concern:`,
            err.response?.data ?? err.message
        );
    }
}

async function createPlatformTicket(ownerToken: string, ranch: RanchConfig) {
    const api = makeApi(ownerToken);

    try {
        await api.post(`/ranches/${ranch.slug}/platform-tickets`, {
            title: `Support request for ${ranch.slug}`,
            description:
                "Please review the ranch setup and confirm the support tools are working properly.",
            category: "support",
            priority: "medium",
        });
        console.log(`✅ [${ranch.slug}] Owner created platform ticket`);
    } catch (err: any) {
        console.log(
            `❌ [${ranch.slug}] Failed platform ticket:`,
            err.response?.data ?? err.message
        );
    }
}

async function seedOneRanch(ranch: RanchConfig) {
    console.log(`\n🚜 Seeding via API for ${ranch.slug}`);

    const members = await getRanchMembers(ranch.ownerToken, ranch.slug);
    console.log(`👥 [${ranch.slug}] Loaded ${members.length} ranch members`);

    const animals = await createAnimals(ranch.ownerToken, ranch);
    await createHealthRecords(ranch.vetToken, ranch, animals);
    await createInventoryItems(ranch.storekeeperToken, ranch);
    await createTasks(ranch, ranch.ownerToken, ranch.managerToken, members);
    await createConcern(ranch.workerToken, ranch);
    await createPlatformTicket(ranch.ownerToken, ranch);

    console.log(`🎉 [${ranch.slug}] API bootstrap completed`);
}

async function main() {
    try {
        for (const ranch of ranches) {
            await seedOneRanch(ranch);
        }

        console.log("\n✅ Both ranch API bootstraps completed successfully");
    } catch (err: any) {
        console.error("BOOTSTRAP_ERROR:", err.response?.data ?? err.message);
    }
}

main();
